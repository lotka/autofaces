\begin{equation} \label{eq:lrn}
  \mathbf{a}(\mathbf{x})_{ijk}
  = \mathbf{b}(\mathbf{x})_{ijk}/\left (\rho + \sum^{\min(N-1,i+n/2)}_{m=\max(0,i-n/2)}\mathbf{b}(\mathbf{x})_{ijm} \right )^\beta
\end{equation}

\center{ $\rho=2,n=5,\alpha=10^{-4}$, $\beta=0.75$. }
